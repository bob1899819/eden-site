SQL Supabase — À exécuter une seule fois
-- EDEN Admin — Structure base de données
-- Collez ce SQL dans l'éditeur SQL de votre projet Supabase

CREATE TABLE IF NOT EXISTS reservations (
  id           UUID DEFAULT gen_random_uuid() PRIMARY KEY,
  created_at   TIMESTAMPTZ DEFAULT NOW(),
  nom          TEXT NOT NULL,
  email        TEXT,
  telephone    TEXT,
  formule      TEXT NOT NULL,
  date_seance  DATE NOT NULL,
  heure        TEXT,
  nb_personnes INTEGER DEFAULT 1,
  extras       JSONB DEFAULT '[]',
  montant_total NUMERIC(10,2) DEFAULT 0,
  statut       TEXT DEFAULT 'pending'
                 CHECK (statut IN ('pending','confirmed','cancelled','done')),
  notes        TEXT
);

CREATE TABLE IF NOT EXISTS demandes (
  id                UUID DEFAULT gen_random_uuid() PRIMARY KEY,
  created_at        TIMESTAMPTZ DEFAULT NOW(),
  nom               TEXT NOT NULL,
  email             TEXT NOT NULL,
  telephone         TEXT,
  formule_souhaitee TEXT,
  date_souhaitee    DATE,
  nb_personnes      INTEGER DEFAULT 1,
  message           TEXT,
  statut            TEXT DEFAULT 'new'
                      CHECK (statut IN ('new','processed','archived'))
);

CREATE TABLE IF NOT EXISTS stocks (
  id            UUID DEFAULT gen_random_uuid() PRIMARY KEY,
  categorie     TEXT NOT NULL,
  nom           TEXT NOT NULL,
  quantite      INTEGER DEFAULT 0,
  unite         TEXT DEFAULT 'unité(s)',
  seuil_alerte  INTEGER DEFAULT 5,
  updated_at    TIMESTAMPTZ DEFAULT NOW()
);

CREATE TABLE IF NOT EXISTS parrainages (
  id                UUID DEFAULT gen_random_uuid() PRIMARY KEY,
  created_at        TIMESTAMPTZ DEFAULT NOW(),
  code              TEXT UNIQUE NOT NULL,
  createur_nom      TEXT NOT NULL,
  filleuls          JSONB DEFAULT '[]',
  utilisations      INTEGER DEFAULT 0,
  reduction_offerte NUMERIC(10,2) DEFAULT 0,
  actif             BOOLEAN DEFAULT TRUE
);

-- Sécurité (Row Level Security)
ALTER TABLE reservations  ENABLE ROW LEVEL SECURITY;
ALTER TABLE demandes       ENABLE ROW LEVEL SECURITY;
ALTER TABLE stocks         ENABLE ROW LEVEL SECURITY;
ALTER TABLE parrainages    ENABLE ROW LEVEL SECURITY;

CREATE POLICY "anon_all" ON reservations  FOR ALL USING (true) WITH CHECK (true);
CREATE POLICY "anon_all" ON demandes       FOR ALL USING (true) WITH CHECK (true);
CREATE POLICY "anon_all" ON stocks         FOR ALL USING (true) WITH CHECK (true);
CREATE POLICY "anon_all" ON parrainages    FOR ALL USING (true) WITH CHECK (true);
